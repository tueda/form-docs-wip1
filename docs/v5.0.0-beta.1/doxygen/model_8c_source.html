2 #[ Explanations :
3
4 This is a second generation much simplified version of model.c
5 Syntax:
6 Model QED;
7 Particle electron,positron,-2;
8 Particle photon,+3;
9 Vertex electron,positron,photon: g;
10 EndModel;
11 or:
12 Model QCD;
13 Particle quark,antiquark,-2;
14 Particle ghost,antighost,-1;
15 Particle gluon,+3;
16 Vertex quark,antiquark,gluon: g;
17 Vertex qhost,antighost,gluon: g;
18 Vertex gluon,gluon,gluon: g;
19 Vertex gluon,gluon,gluon,gluon: g^2;
20 EndModel;
21 Remarks:
22 1: if no second particle is given, a particle and its antiparticle are the same
23 2: Spin statistics is by dimension of SU(2) representation with a sign.
24 3: In a vertex we need to mention the coupling constants.
25 4: Internally a particle gives a vertex with only two lines.
26 5: All particles must have been declared before any vertex.
27
28 The diagrams should be provided as a collection of nodes and edges
29 with momenta with a direction substituted. The other parameters (like
30 Lorentz indices, color indices etc.) can then be provided in a procedure
31 that should go with this mode definition.
32 In principle the edges are not needed if there are no 'insertions', but
33 because we would like to have the insertions as well we need the edges.
34
35 #] Explanations :
36 #[ Includes : model.c
37*/

46VERTEX *CreateVertex(MODEL *m)
47{
48 VERTEX *v, **new;
49 WORD newsize;
50 int i;
51 if ( m->invertices >= m->sizevertices ) {
52 if ( m->sizevertices == 0 ) newsize = 20;
53 else newsize = 2*m->sizevertices;
54 new = (VERTEX **)Malloc1(newsize*sizeof(VERTEX *),"m->vertices");
55 for ( i = 0; i < m->sizevertices; i++ ) new[i] = m->vertices[i];
56 if ( m->sizevertices > 0 ) M_free(m->vertices,"m->vertices");
57 m->vertices = new;
58 m->sizevertices = newsize;
59 }
60 v = (VERTEX *)Malloc1(sizeof(VERTEX),"VERTEX");
61 m->vertices[m->invertices++] = v;
62 v->nparticles = 0;
63 v->ncouplings = 0;
64 v->type = 0;
65 v->error = 0;
66 v->externonly = 0;
67 v->spare = 0;
68 return(v);
69}

76UBYTE *ReadParticle(UBYTE *s, VERTEX *v, MODEL *m, int par)
77{
78 UBYTE *name, c;
79 PARTICLE *p = v->particles + v->nparticles++;
80 WORD type, funnum;
81 int i, j;
82 SKIPBLANKS(s)
83 name = s; s = SkipName(s); c = *s; *s = 0;
84 if ( GetVar(name,&type,&funnum,CFUNCTION,NOAUTO) == NAMENOTFOUND ) {
85 p->number = AddFunction(name,0,VERTEXFUNCTION,0,0,0,-1,-1) + FUNCTION;
86 }
87 else if ( par == 1 && type == CFUNCTION && functions[funnum].spec == VERTEXFUNCTION ) {
88 p->number = funnum+FUNCTION;
89 }
90 else if ( par == 0 && type == CFUNCTION && functions[funnum].spec == VERTEXFUNCTION ) {
91/*
92 We should check whether this particle exists already in this model.
93 If so, this will be an error.
94*/
95 for ( i = 0; i < m->nparticles-1; i++ ) {
96 for ( j = 0; j < m->vertices[i]->nparticles; j++ ) {
97 if ( m->vertices[i]->particles[j].number == funnum ) {
98 MesPrint("&Illegal attempt to redefine a particle in the same model: %s",name);
99 v->error = 1;
100 }
101 }
102 }
103 p->number = funnum+FUNCTION;
104 }
105 else {
106 MesPrint("&Name of particle previously declared as another variable: %s",name);
107 v->error = 1;
108 }
109 *s = c;
110 while ( *s == ',' || *s == ' ' || *s == '\t' ) s++;
111 return(s);
112}

119int CoModel(UBYTE *s)
120{
121 UBYTE *name, c;
122 MODEL *m = (MODEL *)Malloc1(sizeof(MODEL),"Model");
123 WORD numberofset, *e;
124 SETS set;
125 SKIPBLANKS(s)
126 AC.ModelLevel++;
127 int i;
128/*
129 We now have to add the model to the list of models.
130*/
131 if ( AC.modelspace == 0 || AC.nummodels >= AC.modelspace ) {
132 int newspace = 2*AC.modelspace+2, i;
133 MODEL **models = (MODEL **)Malloc1((sizeof(MODEL *))*newspace,"AC.models");
134 for ( i = 0; i < AC.modelspace; i++ ) models[i] = AC.models[i];
135 if ( AC.models ) M_free(AC.models,"AC.models");
136 AC.modelspace = newspace;
137 AC.models = models;
138 }
139
140 AC.models[AC.nummodels++] = m;
141
142 m->vertices = 0;
143 m->couplings = 0;
144 m->nparticles = m->nvertices = m->sizevertices = m->invertices = 0;
145 m->sizecouplings = 0;
146 m->error = 0;
147 m->grccmodel = NULL;
148
149 name = s;
150 if ( FG.cTable[*s] == 0 ) {
151 while ( FG.cTable[*s] <= 1 ) s++;
152 c = *s; *s = 0;
153 m->name = strDup1(name,"Model name");
154 *s = c;
155 SKIPBLANKS(s)
156 if ( *s != 0 ) {
157 MesPrint("&Illegal option in model statement: %s",s);
158 return(1);
159 }
160 }
161 else {
162 m->name = strDup1((UBYTE *)"---","Model name");
163 m->error = 1;
164 MesPrint("&Illegal name for model: %s",name);
165 return(1);
166 }
167/*
168 Now make it a set with one element. The type of the set is rather special
169*/
170 if ( GetName(AC.varnames,m->name,&numberofset,NOAUTO) != NAMENOTFOUND ) {
171 MesPrint("&Name conflict with name %s of model",m->name);
172 return(1);
173 }
174 numberofset = AddSet(name,0);
175 set = Sets + numberofset;
176 set->type = CMODEL;
177 e = (WORD *)FromVarList(&AC.SetElementList);
178 *e = AC.nummodels-1;
179 set->last++;
180 AC.SetList.numtemp = AC.SetList.num;
181 AC.SetElementList.numtemp = AC.SetElementList.num;
182
183 for ( i = 0; i <= MAXLEGS; i++ ) m->legcouple[i] = 0;
184 m->legcouple[2] = 1;
185
186 return(0);

194int CoParticle(UBYTE *s)
195{
196 MODEL *m;
197 VERTEX *v;
198 int i;
199 if ( AC.nummodels <= 0 ) {
200 MesPrint("&No open model for particle statement");
201 return(1);
202 }
203 m = AC.models[AC.nummodels-1];
204 if ( m->nvertices > 0 ) {
205 MesPrint("&In a model description Particle statements should be before Vertex statements.");
206 m->error = 1;
207 return(1);
208 }
209 v = CreateVertex(m);
210 m->nparticles++;
211 s = ReadParticle(s,v,m,0);
212 v->particles[0].type = 1;
213 while ( *s == ',' || *s == ' ' || *s == '\t' ) s++;
214 if ( v->error == 0 ) {
215 if ( FG.cTable[*s] == 0 ) {
216 s = ReadParticle(s,v,m,0);
217 v->particles[1].type = -1;
218 if ( v->particles[0].number == v->particles[1].number ) {
219 v->particles[0].type = 0;
220 v->particles[1].type = 0;
221 }
222 }
223 else { /* Particle = antiparticle */
224 v->particles[1] = v->particles[0];
225 v->nparticles++;
226 v->particles[0].type = 0;
227 v->particles[1].type = 0;
228 }
229 }
230 if ( v->error == 0 && ( *s == '+' || *s == '-' ) ) {
231 int x = 0, sign = ( *s == '-' ) ? -1: 1;
232 s++;
233 while ( FG.cTable[*s] == 1 ) { x = 10*x + *s++ - '0'; }
234 if ( x == 0 ) {
235 v->error = 1;
236 MesPrint("&Spin goes by dimension of SU(2) representation. Zero is not allowed.");
237 }
238 else { v->particles[0].spin = v->particles[1].spin = sign*x; }
239 SKIPBLANKS(s)
240 }
241 else if ( v->error == 0 ) {
242 v->particles[0].spin = v->particles[1].spin = 1;
243 v->particles[0].type = 0;
244 v->particles[1].type = 0;
245 }
246/*
247 Here will come future options
248*/
249 while ( *s == ',' || *s == ' ' || *s == '\t' ) s++;
250 while ( v->error == 0 && *s != 0 && FG.cTable[*s] == 0 ) {
251 UBYTE *opt = s, c;
252 while ( FG.cTable[*s] == 0 ) s++;
253 c = *s; *s = 0;
254 if ( StrICont(opt,(UBYTE *)"external") == 0 ) { v->externonly = 1; }
255 else {
256 MesPrint("&Unrecognized option %s in particle statement.",opt);
257 v->error = 1;
258 }
259 *s = c;
260 while ( *s == ',' || *s == ' ' || *s == '\t' ) s++;
261 }
262/*
263 The couplings array will be used for registering in what kind of vertices the
264 particle is involved. (example: in QCD the quarks only in 2 and 3-point vertices)
265*/
266 for ( i = 0; i < MAXPARTICLES; i++ ) v->couplings[i] = 0;
267 v->couplings[2] = 1;
268 return(v->error);
269}

276int CoVertex(UBYTE *s)
277{
278 UBYTE *ss, *name, c;
279 WORD type;
280 MODEL *m;
281 VERTEX *v;
282 if ( AC.ModelLevel <= 0 ) {
283 MesPrint("&No open model for vertex statement");
284 return(1);
285 }
286 m = AC.models[AC.nummodels-1];
287/*
288 Get an object of type VERTEX
289*/
290 v = CreateVertex(m);
291 m->nvertices++;
292 SKIPBLANKS(s);
293 while ( *s && *s != ':' ) {
294 if ( v->error == 0 ) {
295 s = ReadParticle(s,v,m,1);
296 if ( v->error ) m->error = 1;
297 }
298 else {
299 while ( *s && *s != ':' ) {
300 if ( *s == '[' ) { SKIPBRA1(s) }
301 else if ( *s == '(' ) { SKIPBRA3(s) }
302 else if ( *s == 0 ) {
303 v->error = 1;
304 MesPrint("&No coupling constant in vertex statement.");
305 break;
306 }
307 }
308 break;
309 }
310 }
311 if ( v->error == 0 ) {
312 if ( *s == ':' ) { /* read symbols and powers */
313 s++; SKIPBLANKS(s);
314 while ( *s ) {
315 if ( v->ncouplings >= 2*MAXCOUPLINGS ) {
316 MesPrint("&More than %d coupling constants in vertex.",(WORD)MAXCOUPLINGS);
317 MesPrint(" Recompile with a larger value for MAXCOUPLINGS.");
318 return(1);
319 }
320 ss = s; s = SkipName(s); c = *s; *s = 0; name = ConstructName(ss,0);
321 if ( GetVar(name,&type,&v->couplings[v->ncouplings],CSYMBOL,WITHAUTO)
322 == NAMENOTFOUND ) {
323 WORD minpow = -MAXPOWER;
324 WORD maxpow = MAXPOWER;
325 WORD cplx = 0, dim = 0;
326 v->couplings[v->ncouplings] = AddSymbol(name,minpow,maxpow,cplx,dim);
327 }
328 *s = c;
329 v->ncouplings++;
330/*
331 Now possibly a power
332*/
333 if ( *s == '^' ) { /* we need an integer */
334 WORD x = 0; WORD sign = 1;
335 s++;
336 while ( *s == '-' || *s == '+' ) {
337 if ( *s == '-' ) sign = -sign;
338 s++;
339 }
340 while ( FG.cTable[*s] == 1 ) x = 10*x + *s++ - '0';
341 v->couplings[v->ncouplings++] = x;
342 }
343 else {
344 v->couplings[v->ncouplings++] = 1;
345 }
346 SKIPBLANKS(s);
347 if ( *s == '*' ) { s++; continue; }
348 if ( *s != ',' ) break;
349 s++;
350 SKIPBLANKS(s);
351 }
352 }
353 else {
354 v->error = 1;
355 MesPrint("&A vertex statement needs at least one coupling constant.");
356 }
357 }
358 if ( v->error == 0 ) {
359/*
360 Register which types of vertices are possible for each particle.
361*/
362 int i, j;
363 for ( i = 0; i < v->nparticles; i++ ) {
364 for ( j = 0; j < m->nparticles; j++ ) {
365 if ( m->vertices[j]->particles[0].number == v->particles[i].number ) break;
366 if ( m->vertices[j]->particles[1].number == v->particles[i].number ) break;
367 }
368 m->vertices[j]->couplings[v->nparticles] = 1;
369 }
370 }
371 return(v->error);
372}

379int CoEndModel(UBYTE *s)
380{
381 int i, j, k, kk;
382 WORD csize = 0, *newcouplings, newsize;
383 MODEL *m;
384 VERTEX *v;
385 if ( AC.ModelLevel <= 0 ) {
386 MesPrint("&EndModel statement without matching Model statement");
387 return(1);
388 }
389 m = AC.models[AC.nummodels-1];
390/*
391 Now create a list of all coupling constants.
392 Note that we do not expect an astronomical number of them and hence
393 we use a simple insertion algorithm.
394*/
395 m->ncouplings = 0;
396 for ( i = 0; i < m->nvertices; i++ ) {
397 v = m->vertices[i+m->nparticles];
398 m->legcouple[v->nparticles] = 1;
399 if ( m->ncouplings + v->ncouplings > csize ) {
400 if ( csize == 0 ) newsize = v->ncouplings + 10;
401 else newsize = 2*csize;
402 newcouplings = (WORD *)Malloc1(newsize*sizeof(WORD),"m->couplings");
403 for ( k = 0; k < m->ncouplings; k++ ) newcouplings[k] = m->couplings[k];
404 if ( csize > 0 ) M_free(m->couplings,"m->couplings");
405 m->couplings = newcouplings;
406 csize = newsize;
407 }
408 for ( j = 0; j < v->ncouplings; j += 2 ) {
409 WORD sym = v->couplings[j];
410 for ( k = 0; k < m->ncouplings; k++ ) {
411 if ( sym == m->couplings[k] ) break;
412 if ( sym < m->couplings[k] ) {
413 for ( kk = m->ncouplings; kk > k; k-- )
414 m->couplings[kk] = m->couplings[kk-1];
415 m->couplings[k] = sym;
416 m->ncouplings++;
417 break;
418 }
419 }
420 if ( k >= m->ncouplings ) m->couplings[m->ncouplings++] = sym;
421 }
422 }
423 AC.ModelLevel--;
424 DUMMYUSE(s)
425 return(LoadModel(m));
426}
