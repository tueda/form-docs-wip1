352/*
353 * Macros to be inserted at the end of a structure to align the whole structure.
354 *
355 * In the currently available systems,
356 * sizeof(POSITION) >= sizeof(pointers) == sizeof(LONG) >= sizeof(int)
357 * >= sizeof(WORD) >= sizeof(UBYTE) = 1.
358 * (POSITION is defined in struct.h and contains only an off_t variable.)
359 * Thus, if we put members of a structure in this order and use those macros,
360 * then we can align the data without relying on extra paddings added by
361 * the compiler. For example,
362 * typedef struct {
363 * int *a;
364 * LONG b;
365 * WORD c[2];
366 * UBYTE d;
367 * PADPOINTER(1,0,2,1);
368 * } A;
369 * typedef struct {
370 * POSITION p;
371 * A a; // aligned same as pointers
372 * int *b;
373 * LONG c;
374 * UBYTE d;
375 * PADPOSITION(1,1,0,0,1+sizeof(A));
376 * } B;
377 * The cost for the use of those PADXXX macros is a padding (>= 1 byte) will
378 * be always inserted even in the case that no padding is actually needed.
379 *
380 * Numbers for the arguments have to be calculated manually and so very
381 * error-prone. Be careful!
382 *
383 * Note that there is a 32-bit system in which off_t is aligned on 8-byte
384 * boundary, (e.g., Cygwin with large file support), but still the above
385 * inequalities are satisfied.
386 *
387 * The legendary story of these macros--they fixed some problems in ancient
388 * times when compilers were unreliable and didn't know how to correctly compute
389 * structure paddings--has been handed down, though nowadays there are only
390 * disadvantages for them in practice (ancient compilers most likely can't
391 * compile C99 and C++98+TR1 sources anyway).
392 */
393#define PADDUMMY(type, size) \
394 UBYTE d_u_m_m_y[form_alignof(type) - ((size) & (form_alignof(type) - 1))]
395#define PADPOSITION(ptr_,long_,int_,word_,byte_) \
396 PADDUMMY(off_t, \
397 + sizeof(int *) * (ptr_) \
398 + sizeof(LONG) * (long_) \
399 + sizeof(int) * (int_) \
400 + sizeof(WORD) * (word_) \
401 + sizeof(UBYTE) * (byte_) \
402 )
403#define PADPOINTER(long_,int_,word_,byte_) \
404 PADDUMMY(int *, \
405 + sizeof(LONG) * (long_) \
406 + sizeof(int) * (int_) \
407 + sizeof(WORD) * (word_) \
408 + sizeof(UBYTE) * (byte_) \
409 )
410#define PADLONG(int_,word_,byte_) \
411 PADDUMMY(LONG, \
412 + sizeof(int) * (int_) \
413 + sizeof(WORD) * (word_) \
414 + sizeof(UBYTE) * (byte_) \
415 )
416#define PADINT(word_,byte_) \
417 PADDUMMY(int, \
418 + sizeof(WORD) * (word_) \
419 + sizeof(UBYTE) * (byte_) \
420 )
421#define PADWORD(byte_) \
422 PADDUMMY(WORD, \
423 + sizeof(UBYTE) * (byte_) \
424 )
