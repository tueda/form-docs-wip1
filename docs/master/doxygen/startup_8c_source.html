114static void PrintHeader(int with_full_info)
115{
116#ifdef WITHMPI
117 if ( PF.me == MASTER && !AM.silent ) {
118#else
119 if ( !AM.silent ) {
120#endif
121 char buffer1[250], buffer2[80], *s = buffer1, *t = buffer2;
122 WORD length, n;
123 for ( n = 0; n < 250; n++ ) buffer1[n] = ' ';
124 /*
125 * NOTE: we expect that the compiler optimizes strlen("string literal")
126 * to just a number.
127 */
128 if ( strlen(VERSIONSTR) <= 100 ) {
129 strcpy(s,VERSIONSTR);
130 s += strlen(VERSIONSTR);
131 *s = 0;
132 }
133 else {
134 /*
135 * Truncate when it is too long.
136 */
137 strncpy(s,VERSIONSTR,97);
138 s[97] = '.';
139 s[98] = '.';
140 s[99] = ')';
141 s[100] = '\0';
142 s += 100;
143 }
144/*
145 By now we omit the message about 32/64 bits. It should all be 64.
146
147 s += snprintf(s,250-(s-buffer1)," %d-bits",(WORD)(sizeof(WORD)*16));
148 *s = 0;
149*/
150 if ( with_full_info ) {
151#if defined(WITHPTHREADS) || defined(WITHMPI)
152#if defined(WITHPTHREADS)
153 int nworkers = AM.totalnumberofthreads-1;
154#elif defined(WITHMPI)
155 int nworkers = PF.numtasks-1;
156#endif
157 s += snprintf(s,250-(s-buffer1)," %d worker",nworkers);
158 *s = 0;
159/* while ( *s ) s++; */
160 if ( nworkers != 1 ) {
161 *s++ = 's';
162 *s = '\0';
163 }
164#endif
165
166 snprintf(t,80-(t-buffer2),"Run: %s",MakeDate());
167 while ( *t ) t++;
168
169 /*
170 * Align the date to the right, if it fits in a line.
171 */
172 length = (s-buffer1) + (t-buffer2);
173 if ( length+2 <= AC.LineLength ) {
174 for ( n = AC.LineLength-length; n > 0; n-- ) *s++ = ' ';
175 *s = 0;
176 strcat(s,buffer2);
177 while ( *s ) s++;
178 }
179 else {
180 *s = 0;
181 strcat(s," ");
182 while ( *s ) s++;
183 *s = 0;
184 strcat(s,buffer2);
185 while ( *s ) s++;
186 }
187 }
188
189 /*
190 * If the header information doesn't fit in a line, we need to extend
191 * the line length temporarily.
192 */
193 length = s-buffer1;
194 if ( length <= AC.LineLength ) {
195 MesPrint("%s",buffer1);
196 }
197 else {
198 WORD oldLineLength = AC.LineLength;
199 AC.LineLength = length;
200 MesPrint("%s",buffer1);
201 AC.LineLength = oldLineLength;
202 }
203 }
204#ifdef WINDOWS
205 PrintDeprecation("the native Windows version", "issues/623");
206#endif
207#ifdef ILP32
208 PrintDeprecation("the 32-bit version", "issues/624");
209#endif
210#ifdef WITHMPI
211 PrintDeprecation("the MPI version (ParFORM)", "issues/625");
212#endif
213 if ( AC.CheckpointFlag ) {
214 PrintDeprecation("the checkpoint mechanism", "issues/626");
215 }
216}

237int DoTail(int argc, UBYTE **argv)
238{
239 int errorflag = 0, onlyversion = 1;
240 UBYTE *s, *t, *copy;
241 int threadnum = 0;
242 argc--; argv++;
243 AM.ClearStore = 0;
244 AM.TimeLimit = 0;
245 AM.LogType = -1;
246 AM.HoldFlag = AM.qError = AM.Interact = AM.FileOnlyFlag = 0;
247 AM.InputFileName = AM.LogFileName = AM.IncDir = AM.TempDir = AM.TempSortDir =
248 AM.SetupDir = AM.SetupFile = AM.Path = 0;
249 AM.FromStdin = 0;
250 /* Always use MultiRun, "-M" option is now ignored. */
251 AM.MultiRun = 1;
252 if ( argc < 1 ) {
253 onlyversion = 0;
254 goto printversion;
255 }
256 while ( argc >= 1 ) {
257 s = *argv++; argc--;
258 if ( *s == '-' || ( *s == '/' && ( argc > 0 || AM.Interact ) ) ) {
259 s++;
260 switch (*s) {
261 case 'c': /* Error checking only */
262 AM.qError = 1; break;
263 case 'C': /* Next arg is filename of log file */
264 TAKEPATH(AM.LogFileName) break;
265 case 'D':
266 case 'd': /* Next arg is define preprocessor var. */
267 t = copy = strDup1(*argv,"Dotail");
268 while ( *t && *t != '=' ) t++;
269 if ( *t == 0 ) {
270 if ( PutPreVar(copy,(UBYTE *)"1",0,0) < 0 ) return(-1);
271 }
272 else {
273 *t++ = 0;
274 if ( PutPreVar(copy,t,0,0) < 0 ) return(-1);
275 t[-1] = '=';
276 }
277 M_free(copy,"-d prevar");
278 argv++; argc--; break;
279 case 'f': /* Output only to regular log file */
280 AM.FileOnlyFlag = 1; AM.LogType = 0; break;
281 case 'F': /* Output only to log file. Further like L. */
282 AM.FileOnlyFlag = 1; AM.LogType = 1; break;
283 case 'h': /* For old systems: wait for key before exit */
284 AM.HoldFlag = 1; break;
285 case 'i':
286 if ( StrCmp(s, (UBYTE *)"ignore-deprecation") == 0 ) {
287 AM.IgnoreDeprecation = 1;
288 break;
289 }
290#ifdef WITHINTERACTION
291 /* Interactive session (not used yet) */
292 AM.Interact = 1;
293 break;
294#else
295 goto IllegalOption;
296#endif
297 case 'I': /* Next arg is dir for inc/prc/sub files */
298 TAKEPATH(AM.IncDir) break;
299 case 'l': /* Make regular log file */
300 if ( s[1] == 'l' ) AM.LogType = 1; /*compatibility! */
301 else AM.LogType = 0;
302 break;
303 case 'L': /* Make log file with only final statistics */
304 AM.LogType = 1; break;
305 case 'M': /* Multirun. Name of tempfiles will contain PID */
306 /* This option is now ignored. We always use MultiRun. */
307 /* AM.MultiRun = 1; */
308 break;
309 case 'm': /* Read number of threads */
310 case 'w': /* Read number of workers */
311 t = s++;
312 threadnum = 0;
313 while ( *s >= '0' && *s <= '9' )
314 threadnum = 10*threadnum + *s++ - '0';
315 if ( *s ) {
316#ifdef WITHMPI
317 if ( PF.me == MASTER )
318#endif
319 printf("Illegal value for option m or w: %s\n",t);
320 errorflag++;
321 }
322/* if ( threadnum == 1 ) threadnum = 0; */
323 threadnum++;
324 break;
325 case 'W': /* Print the wall-clock time on the master. */
326 AM.ggWTimeStatsFlag = 1;
327 break;
328/*
329 case 'n':
330 Reserved for number of slaves without MPI
331*/
332 case 'p':
333#ifdef WITHEXTERNALCHANNEL
334 /*There are two possibilities: -p|-pipe*/
335 if(s[1]=='i'){
336 if( (s[2]=='p')&&(s[3]=='e')&&(s[4]=='\0') ){
337 argc--;
338 /*Initialize pre-set external channels, see
339 the file extcmd.c:*/
340 if(initPresetExternalChannels(*argv++,AX.timeout)<1){
341#ifdef WITHMPI
342 if ( PF.me == MASTER )
343#endif
344 printf("Error initializing preset external channels\n");
345 errorflag++;
346 }
347 AX.timeout=-1;/*This indicates that preset channels
348 are initialized from cmdline*/
349 }else{
350#ifdef WITHMPI
351 if ( PF.me == MASTER )
352#endif
353 printf("Illegal option in call of FORM: %s\n",s);
354 errorflag++;
355 }
356 }else
357#else
358 if ( s[1] ) {
359 if ( ( s[1]=='i' ) && ( s[2] == 'p' ) && (s[3] == 'e' )
360 && ( s[4] == '\0' ) ){
361#ifdef WITHMPI
362 if ( PF.me == MASTER )
363#endif
364 printf("Illegal option: Pipes not supported on this system.\n");
365 }
366 else {
367#ifdef WITHMPI
368 if ( PF.me == MASTER )
369#endif
370 printf("Illegal option: %s\n",s);
371 }
372 errorflag++;
373 }
374 else
375#endif
376 {
377 /* Next arg is a path variable like in environment */
378 TAKEPATH(AM.Path)
379 }
380 break;
381 case 'q': /* Quiet option. Only output. Same as -si */
382 AM.silent = 1; break;
383 case 'R': /* recover from saved snapshot */
384 AC.CheckpointFlag = -1;
385 break;
386 case 's': /* Next arg is dir with form.set to be used */
387 if ( ( s[1] == 'o' ) && ( s[2] == 'r' ) && ( s[3] == 't' ) ) {
388 if(s[4]== '=' ) {
389 AM.TempSortDir = s+5;
390 }
391 else {
392 AM.TempSortDir = *argv++;
393 argc--;
394 }
395 }
396 else if ( s[1] == 'i' ) { /* compatibility: silent/quiet */
397 AM.silent = 1;
398 }
399 else {
400 TAKEPATH(AM.SetupDir)
401 }
402 break;
403 case 'S': /* Next arg is setup file */
404 TAKEPATH(AM.SetupFile) break;
405 case 't': /* Next arg is directory for temp files */
406 if ( s[1] == 's' ) {
407 s++;
408 AM.havesortdir = 1;
409 TAKEPATH(AM.TempSortDir)
410 }
411 else {
412 TAKEPATH(AM.TempDir)
413 }
414 break;
415 case 'T': /* Print the total size used at end of job */
416 AM.PrintTotalSize = 1; break;
417 case 'v':
418printversion:;
419#ifdef WITHMPI
420 if ( PF.me == MASTER )
421#endif
422 PrintHeader(0);
423 if ( onlyversion ) return(1);
424 goto NoFile;
425 case 'y': /* Preprocessor dumps output. No compilation. */
426 AP.PreDebug = PREPROONLY; break;
427 case 'z': /* The number following is a time limit in sec. */
428 t = s++;
429 AM.TimeLimit = 0;
430 while ( *s >= '0' && *s <= '9' )
431 AM.TimeLimit = 10*AM.TimeLimit + *s++ - '0';
432 break;
433 case 'Z': /* Removes the .str file on crash, no matter its contents */
434 AM.ClearStore = 1; break;
435 case '\0': /* "-" to use STDIN for the input. */
436#ifdef WITHMPI
437 /* At the moment, ParFORM doesn't implement STDIN broadcasts. */
438 if ( PF.me == MASTER )
439 printf("Sorry, reading STDIN as input is currently not supported by ParFORM\n");
440 errorflag++;
441#endif
442 AM.FromStdin = 1;
443 AC.NoShowInput = 1; // disable input echoing by default
444 break;
445 default:
446 if ( FG.cTable[*s] == 1 ) {
447 AM.SkipClears = 0; t = s;
448 while ( FG.cTable[*t] == 1 )
449 AM.SkipClears = 10*AM.SkipClears + *t++ - '0';
450 if ( *t != 0 ) {
451#ifdef WITHMPI
452 if ( PF.me == MASTER )
453#endif
454 printf("Illegal numerical option in call of FORM: %s\n",s);
455 errorflag++;
456 }
457 }
458 else {
459IllegalOption:
460#ifdef WITHMPI
461 if ( PF.me == MASTER )
462#endif
463 printf("Illegal option in call of FORM: %s\n",s);
464 errorflag++;
465 }
466 break;
467 }
468 }
469 else if ( argc == 0 && !AM.Interact ) AM.InputFileName = argv[-1];
470 else {
471#ifdef WITHMPI
472 if ( PF.me == MASTER )
473#endif
474 printf("Illegal option in call of FORM: %s\n",s);
475 errorflag++;
476 }
477 }
478 AM.totalnumberofthreads = threadnum;
479 if ( AM.InputFileName ) {
480 if ( AM.FromStdin ) {
481 printf("STDIN and the input filename cannot be specified simultaneously\n");
482 errorflag++;
483 }
484 s = AM.InputFileName;
485 while ( *s ) s++;
486 if ( s < AM.InputFileName+4 ||
487 s[-4] != '.' || s[-3] != 'f' || s[-2] != 'r' || s[-1] != 'm' ) {
488 t = (UBYTE *)Malloc1((s-AM.InputFileName)+5,"adding .frm");
489 s = AM.InputFileName;
490 AM.InputFileName = t;
491 while ( *s ) *t++ = *s++;
492 *t++ = '.'; *t++ = 'f'; *t++ = 'r'; *t++ = 'm'; *t = 0;
493 }
494 if ( AM.LogType >= 0 && AM.LogFileName == 0 ) {
495 AM.LogFileName = strDup1(AM.InputFileName,"name of logfile");
496 s = AM.LogFileName;
497 while ( *s ) s++;
498 s[-3] = 'l'; s[-2] = 'o'; s[-1] = 'g';
499 }
500 }
501#ifdef WITHINTERACTION
502 else if ( AM.Interact ) {
503 if ( AM.LogType >= 0 ) {
504/*
505 We may have to do better than just taking a name.
506 It is not unique! This will be left for later.
507*/
508 AM.LogFileName = deflogname;
509 }
510 }
511#endif
512 else if ( AM.FromStdin ) {
513 /* Do nothing. */
514 }
515 else {
516NoFile:
517#ifdef WITHMPI
518 if ( PF.me == MASTER )
519#endif
520 printf("No filename specified in call of FORM\n");
521 errorflag++;
522 }
523 if ( AM.Path == 0 ) AM.Path = (UBYTE *)getenv("FORMPATH");
524 if ( AM.Path ) {
525 /*
526 * AM.Path is taken from argv or getenv. Reallocate it to avoid invalid
527 * frees when AM.Path has to be changed.
528 */
529 AM.Path = strDup1(AM.Path,"DoTail Path");
530 }
531 return(errorflag);
532}

542int OpenInput(VOID)
543{
544 int oldNoShowInput = AC.NoShowInput;
545 UBYTE c;
546 if ( !AM.Interact ) {
547 if ( AM.FromStdin ) {
548 if ( OpenStream(0,INPUTSTREAM,0,PRENOACTION) == 0 ) {
549 Error0("Cannot open STDIN");
550 return(-1);
551 }
552 }
553 else {
554 if ( OpenStream(AM.InputFileName,FILESTREAM,0,PRENOACTION) == 0 ) {
555 Error1("Cannot open file",AM.InputFileName);
556 return(-1);
557 }
558 if ( AC.CurrentStream->inbuffer <= 0 ) {
559 Error1("No input in file",AM.InputFileName);
560 return(-1);
561 }
562 }
563 AC.NoShowInput = 1;
564 while ( AM.SkipClears > 0 ) {
565 c = GetInput();
566 if ( c == ENDOFINPUT ) {
567 Error0("Not enough .clear instructions in input file");
568 }
569 if ( c == '\\' ) {
570 c = GetInput();
571 if ( c == ENDOFINPUT )
572 Error0("Not enough .clear instructions in input file");
573 continue;
574 }
575 if ( c == ' ' || c == '\t' ) continue;
576 if ( c == '.' ) {
577 c = GetInput();
578 if ( tolower(c) == 'c' ) {
579 c = GetInput();
580 if ( tolower(c) == 'l' ) {
581 c = GetInput();
582 if ( tolower(c) == 'e' ) {
583 c = GetInput();
584 if ( tolower(c) == 'a' ) {
585 c = GetInput();
586 if ( tolower(c) == 'r' ) {
587 c = GetInput();
588 if ( FG.cTable[c] > 2 ) {
589 AM.SkipClears--;
590 }
591 }
592 }
593 }
594 }
595 }
596 while ( c != '\n' && c != '\r' && c != ENDOFINPUT ) {
597 c = GetInput();
598 if ( c == '\\' ) c = GetInput();
599 }
600 }
601 else if ( c == '\n' || c == '\r' ) continue;
602 else {
603 while ( ( c = GetInput() ) != '\n' && c != '\r' ) {
604 if ( c == ENDOFINPUT ) {
605 Error0("Not enough .clear instructions in input file");
606 }
607 }
608 }
609 }
610 AC.NoShowInput = oldNoShowInput;
611 }
612 if ( AM.LogFileName ) {
613#ifdef WITHMPI
614 if ( PF.me != MASTER ) {
615 /*
616 * Only the master writes to the log file. On slaves, we need
617 * a dummy handle, without opening the file.
618 */
619 extern FILES **filelist; /* in tools.c */
620 int i = CreateHandle();
621 RWLOCKW(AM.handlelock);
622 filelist[i] = (FILES *)123; /* Must be nonzero to prevent a reuse in CreateHandle. */
623 UNRWLOCK(AM.handlelock);
624 AC.LogHandle = i;
625 }
626 else
627#endif
628 if ( AC.CheckpointFlag != -1 ) {
629 if ( ( AC.LogHandle = CreateLogFile((char *)(AM.LogFileName)) ) < 0 ) {
630 Error1("Cannot create logfile",AM.LogFileName);
631 return(-1);
632 }
633 }
634 else {
635 if ( ( AC.LogHandle = OpenAddFile((char *)(AM.LogFileName)) ) < 0 ) {
636 Error1("Cannot re-open logfile",AM.LogFileName);
637 return(-1);
638 }
639 }
640 }
641 return(0);
642}

645 #] OpenInput :
646 #[ ReserveTempFiles :
647
648 Order of preference:
649 a: if there is a path in the commandtail, take that.
650 b: if none, try in the form.set file.
651 c: if still none, try in the environment for the variable FORMTMP
652 d: if still none, try the current directory.
653
654 The parameter indicates action in the case of multithreaded running.
655 par = 0 : We just run on a single processor. Keep everything normal.
656 par = 1 : Multithreaded running startup phase 1.
657 par = 2 : Multithreaded running startup phase 2.
658*/

660UBYTE *emptystring = (UBYTE *)".";
661UBYTE *defaulttempfilename = (UBYTE *)"xformxxx.str";
662/* This is the length of the above default, but with 7 spaces for PID digits
663 instead of the "xxx". (Previously FORM used 5 digits and this value was 14) */
664#define DEFAULTFNAMELENGTH 16
665
666VOID ReserveTempFiles(int par)
667{
668 GETIDENTITY
669 SETUPPARAMETERS *sp;
670 UBYTE *s, *t, *tenddir, *tenddir2, c;
671 int i = 0;
672 WORD j;
673 if ( par == 0 || par == 1 ) {
674 if ( AM.TempDir == 0 ) {
675 sp = GetSetupPar((UBYTE *)"tempdir");
676 if ( ( sp->flags & USEDFLAG ) != USEDFLAG ) {
677 AM.TempDir = (UBYTE *)getenv("FORMTMP");
678 if ( AM.TempDir == 0 ) AM.TempDir = emptystring;
679 }
680 else AM.TempDir = (UBYTE *)(sp->value);
681 }
682 if ( AM.TempSortDir == 0 ) {
683 if ( AM.havesortdir ) {
684 sp = GetSetupPar((UBYTE *)"tempsortdir");
685 AM.TempSortDir = (UBYTE *)(sp->value);
686 }
687 else {
688 AM.TempSortDir = (UBYTE *)getenv("FORMTMPSORT");
689 if ( AM.TempSortDir == 0 ) AM.TempSortDir = AM.TempDir;
690 }
691 }
692/*
693 We have now in principle a path but we will use its first element only.
694 Later that should become more complicated. Then we will use a path and
695 when one device is full we can continue on the next one.
696*/
697 s = AM.TempDir; i = 200; /* Some extra for VMS */
698 while ( *s && *s != PATHSEPARATOR ) { if ( *s == '\\' ) s++; s++; i++; }
699 FG.fnamesize = sizeof(UBYTE)*(i+DEFAULTFNAMELENGTH);
700 FG.fname = (char *)Malloc1(FG.fnamesize,"name for temporary files");
701 s = AM.TempDir; t = (UBYTE *)FG.fname;
702 while ( *s && *s != PATHSEPARATOR ) { if ( *s == '\\' ) s++; *t++ = *s++; }
703 if ( (char *)t > FG.fname && t[-1] != SEPARATOR && t[-1] != ALTSEPARATOR )
704 *t++ = SEPARATOR;
705 *t = 0;
706 tenddir = t;
707 FG.fnamebase = t-(UBYTE *)(FG.fname);
708
709 s = AM.TempSortDir; i = 200; /* Some extra for VMS */
710 while ( *s && *s != PATHSEPARATOR ) { if ( *s == '\\' ) s++; s++; i++; }
711
712 FG.fname2size = sizeof(UBYTE)*(i+DEFAULTFNAMELENGTH);
713 FG.fname2 = (char *)Malloc1(FG.fname2size,"name for sort files");
714 s = AM.TempSortDir; t = (UBYTE *)FG.fname2;
715 while ( *s && *s != PATHSEPARATOR ) { if ( *s == '\\' ) s++; *t++ = *s++; }
716 if ( (char *)t > FG.fname2 && t[-1] != SEPARATOR && t[-1] != ALTSEPARATOR )
717 *t++ = SEPARATOR;
718 *t = 0;
719 tenddir2 = t;
720 FG.fname2base = t-(UBYTE *)(FG.fname2);
721
722 t = tenddir;
723 s = defaulttempfilename;
724#ifdef WITHMPI
725 {
726 int iii;
727 iii = snprintf((char*)t,FG.fnamesize-((char*)t-FG.fname),"%d",PF.me);
728 t+= iii;
729 s+= iii; /* in case defaulttmpfilename is too short */
730 }
731#endif
732 while ( *s ) *t++ = *s++;
733 *t = 0;
